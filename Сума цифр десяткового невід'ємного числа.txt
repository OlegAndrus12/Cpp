#include <iostream>
#include <cmath>

using namespace std;

int main() {
	unsigned number;
	cout << "number: ";
	cin >> number;

	unsigned i = number;
	unsigned s = 0;
	do {
		s += i % 10;
		i /= 10;
	} while (i);
	cout << "number= " << number << " sum of digits= " << s << endl;
	system("pause");
}
