#include<iostream>
#include<cmath>
#include <istream>

using namespace std;

void main() {
	const int rows = 5;
	const int columns = 5;
	int matrix[rows][columns];
	cout << "Matrix " << rows << "x" << columns << ":" << endl;
	for (int i = 0; i < rows;++i)
	{
		for (int k = 0; k < columns; ++k)
		{
			cout << "Rows: " << i << " Columns: " << k << endl;
			cin >> matrix[i][k];
		}
	}
	int s[rows];
	for (int k = 0; k < rows;++k)
	{
		s[k] = 0;
		for (int i = 0; i < columns;++i)//сума елементів рядка матриці
		{
			s[k] += matrix[k][i];
			cout <<"S "<<k <<"= "<< s[k] << endl;
		}
		
	}
	system("pause");
}
