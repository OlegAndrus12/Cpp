#include <iostream>
#include <cmath>

using namespace std;

void main() {
	double y, a, y1;
	double S = 0;
	double S1 = 0;
	for (a = 99; a >= 6; a -= 3)
	{
		y = sqrt(a);
		S += y;
		y1 = sqrt((a - 3) + S);
		S1 += y1;
		
	}
	cout << "y= " << S1 << endl;
	system("pause");
}
