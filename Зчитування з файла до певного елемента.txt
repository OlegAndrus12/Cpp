#include <iostream>
#include <fstream>
#include <string>


using namespace std;

int main() {
	ifstream in("text.txt"); // підключаємось до текстового файла (беремо дані із текстового файла)
	string str; // змінна string рядкового типу даних без значення
	char c; // змінна с - елемент із файла
	while(in>>c && c != ';') // з ліва на право зчитується документ, якщо символ не дорівнює ';' ми зараховуємо його до змінної str, якщо при зчитуванні символів, ми дійшли до символа, ';' - зчитування припиняється

	{
		str += c;
	}
	cout << str.length() << endl << str << endl; // довжина рядка до ';' дорівнює кількості цих символів. Виводимо кількість символів до ';' та сам рядок до ';'
	system("pause");

}
