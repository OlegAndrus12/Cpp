#include <iostream>

using namespace std;

void main()
{
	int N;
	cout << "Enter N" << endl;
	cin >> N;
	double* Arr = new double[N];
	for (int i = 0; i < N; ++i)
	{
		cin >> Arr[i];
	}

	double Sum1 = 0;
	for (int i = 0;i < N;++i)
	{
		Sum1 += Arr[i];
	}
	double Sum2 = 0;
	for (int i = 0;i < N;++i)
	{
		Sum2 += *(Arr+i);
	}
	cout << "Sum1 = " << Sum1 << " Sum2 = " << Sum2 << endl;
	delete [] Arr;
	system("pause");
}
